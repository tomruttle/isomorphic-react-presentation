React with native Javascript

  • React.DOM library has all of the HTML elements you need
    • (probably)
  • Intuitive API - it works how you would expect
  • No transpilation required
    • (Though still need to bundle client-side code)
  • Great for prototyping, or for use with external component libraries

const { DOM: r } = require('react');

const SimpleForm = () =>
  r.form(
    { method: 'post', action: '/submit' },
    r.h3({}, 'A simple Form'),
    r.label({ htmlFor: 'text' }, 'Please input some text: '),
    r.input({ type: 'text', name: 'text' }),
    r.button({ type: 'submit' }, 'SUBMIT')
  );

SimpleForm.propTypes = {
  actionEndpoint: React.PropTypes.string.isRequired,
};

module.exports = SimpleForm;
